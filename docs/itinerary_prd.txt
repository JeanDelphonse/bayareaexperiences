1. Feature Overview
1.1 Purpose
This PRD defines the Dynamic Itinerary Generation System for Bay Area Experiences. When a customer books an experience, the system automatically generates a personalized, day-of itinerary that: (1) starts from the customer's selected pickup city, (2) follows the core stops promised in the experience title and description, (3) integrates 1–2 relevant local events or activities happening near the pickup location on the tour date, and (4) is delivered to both the customer on their confirmation page and to the assigned staff member or provider via a dedicated mobile briefing page.
1.2 The Problem Being Solved
Currently, all customers booking the same experience receive an identical description regardless of where they are departing from or what is happening locally on their tour date. A customer picking up from San Jose sees the same itinerary as one picking up from San Francisco. There is no personalization, no local context, and no day-of relevance. The Dynamic Itinerary System changes this — each booking gets a unique, date-specific, location-aware itinerary that makes the experience feel curated specifically for that guest.

1.4 Itinerary Structure
Every generated itinerary contains the following sections, regardless of experience or pickup city:
Header — experience name, tour date, pickup city, pickup time, guest name, guest count
Pickup details — exact pickup address (from booking), estimated departure time
Local Context (1–2 items) — events or activities happening near the pickup city on the tour date with a note on how they relate to the day
Experience Stops — ordered list of the core stops for the booked experience, each with: stop name, estimated arrival time, what to expect, approximate time at stop
Travel notes — estimated drive times between stops from the pickup city (may differ from standard route)
Inclusions reminder — snacks, water, what is included in the booking
Drop-off details — estimated return time, drop-off location
Customer notes — any special requests entered at booking (dietary needs, accessibility, occasion)

# app/itinerary/events.py — Ticketmaster fetch
import requests, os
from datetime import datetime, timedelta

TM_BASE = 'https://app.ticketmaster.com/discovery/v2/events.json'

# Segment names to include (exclude heavy metal, adult content, etc.)
RELEVANT_SEGMENTS = ['Music', 'Arts & Theatre', 'Family', 'Sports', 'Miscellaneous']

# Event categories to exclude per experience type
EXCLUDE_FOR_EXPERIENCE = {
    'airport-transfer':     ['Sports'],        # Don't suggest stadium events for airport runs
    'corporate-team-outing':['Family'],         # Avoid kids events for corporate bookings
    'celebration-tour':     [],                 # All events welcome for celebrations
}

def fetch_ticketmaster_events(city: str, state_code: str, tour_date: str,
                              experience_slug: str = None) -> list:
    """
    Fetch up to 5 local events from Ticketmaster for the given city and date.
    tour_date format: 'YYYY-MM-DD'
    Returns list of dicts: {name, venue, date, category, url}
    """
    api_key = os.environ.get('TICKETMASTER_API_KEY', '')
    if not api_key:
        return []

    # Search window: same day only (start of day to end of day)
    start = f'{tour_date}T00:00:00Z'
    end   = f'{tour_date}T23:59:59Z'

    params = {
        'apikey':         api_key,
        'city':           city,
        'stateCode':      state_code,
        'startDateTime':  start,
        'endDateTime':    end,
        'size':           10,
        'sort':           'relevance,desc',
    }

    try:
        resp = requests.get(TM_BASE, params=params, timeout=8)
        resp.raise_for_status()
        data = resp.json()
    except Exception as e:
        import logging
        logging.getLogger('itinerary').warning(f'Ticketmaster fetch failed: {e}')
        return []

    events = []
    for ev in data.get('_embedded', {}).get('events', []):
        segment = ''
        try:
            segment = ev['classifications'][0]['segment']['name']
        except (KeyError, IndexError):
            pass

        # Skip irrelevant segments
        if segment and segment not in RELEVANT_SEGMENTS:
            continue

        # Skip excluded categories for this experience type
        exclude = EXCLUDE_FOR_EXPERIENCE.get(experience_slug, [])
        if segment in exclude:
            continue

        venue_name = ''
        venue_city = ''
        try:
            venue_name = ev['_embedded']['venues'][0]['name']
            venue_city = ev['_embedded']['venues'][0]['city']['name']
        except (KeyError, IndexError):
            pass

        events.append({
            'source':   'ticketmaster',
            'name':     ev.get('name', ''),
            'venue':    venue_name,
            'city':     venue_city or city,
            'date':     ev.get('dates', {}).get('start', {}).get('localDate', tour_date),
            'category': segment,
            'url':      ev.get('url', ''),
        })

        if len(events) >= 5:
            break

    return events

# app/itinerary/events.py (continued)

EB_BASE = 'https://www.eventbriteapi.com/v3/events/search/'

# Eventbrite category IDs most relevant to BAE experiences
EB_CATEGORIES = '103,110,113,101,104'  # Arts, Food, Community, Music, Outdoors

def fetch_eventbrite_events(city: str, state_code: str, tour_date: str) -> list:
    """
    Fetch local community events from Eventbrite.
    Falls back gracefully if API key not set.
    """
    api_key = os.environ.get('EVENTBRITE_API_KEY', '')
    if not api_key:
        return []

    address = f'{city}, {state_code}'
    start   = f'{tour_date}T00:00:00'
    end     = f'{tour_date}T23:59:59'

    headers = {'Authorization': f'Bearer {api_key}'}
    params  = {
        'location.address':         address,
        'location.within':          '20mi',
        'start_date.range_start':   start,
        'start_date.range_end':     end,
        'categories':               EB_CATEGORIES,
        'expand':                   'venue',
        'page_size':                5,
        'sort_by':                  'best',
    }

    try:
        resp = requests.get(EB_BASE, headers=headers, params=params, timeout=8)
        resp.raise_for_status()
        data = resp.json()
    except Exception as e:
        import logging
        logging.getLogger('itinerary').warning(f'Eventbrite fetch failed: {e}')
        return []

    events = []
    for ev in data.get('events', []):
        venue_name = ''
        try:
            venue_name = ev['venue']['name']
        except (KeyError, TypeError):
            pass

        events.append({
            'source':   'eventbrite',
            'name':     ev.get('name', {}).get('text', ''),
            'venue':    venue_name,
            'city':     city,
            'date':     tour_date,
            'category': 'Community Event',
            'url':      ev.get('url', ''),
        })

    return events


def get_local_events(city: str, state_code: str, tour_date: str,
                      experience_slug: str = None) -> list:
    """
    Combine Ticketmaster + Eventbrite results, deduplicate, return up to 5.
    Always returns a list (empty if both APIs fail) — never raises.
    """
    tm_events = fetch_ticketmaster_events(city, state_code, tour_date, experience_slug)
    eb_events = fetch_eventbrite_events(city, state_code, tour_date)

    # Combine: Ticketmaster first, then Eventbrite
    combined = tm_events + eb_events

    # Simple dedup by lowercase name
    seen   = set()
    unique = []
    for ev in combined:
        key = ev['name'].lower().strip()
        if key and key not in seen:
            seen.add(key)
            unique.append(ev)

    return unique[:5]  # Cap at 5 candidates for Claude

3. Claude AI — Itinerary Generation
3.1 Architecture Decision
Claude (claude-sonnet-4-6) is called once per itinerary generation. It receives a structured prompt containing all inputs and returns a complete itinerary as a JSON object. The temperature is set low (0.4) for consistency and factual grounding. Claude does not invent stops, drive times, or local facts — it structures and personalizes the content from the verified data provided in the prompt.
3.2 Prompt Structure

3.3 Generator Function
# app/itinerary/generator.py (continued)
from datetime import datetime, timezone
from app.models import Experience, Booking, Timeslot

def generate_itinerary(booking: Booking) -> dict:
    """
    Generate a personalized itinerary for a booking.
    Returns the itinerary as a Python dict.
    Never raises — returns a fallback dict on any failure.
    """
    try:
        experience = booking.experience
        timeslot   = booking.timeslot
        tour_date  = str(timeslot.slot_date)
        start_time = timeslot.start_time.strftime('%I:%M %p') if timeslot.start_time else '9:00 AM'
        pickup_city = booking.pickup_city or 'San Francisco'
        state_code  = 'CA'

        # 1. Fetch local events from Ticketmaster + Eventbrite
        local_events = get_local_events(
            city=pickup_city,
            state_code=state_code,
            tour_date=tour_date,
            experience_slug=experience.slug,
        )

        # Format local events for the prompt
        if local_events:
            local_events_text = '\n'.join([
                f'- {ev["name"]} at {ev["venue"]} ({ev["category"]}) — {ev["url"]}'
                for ev in local_events[:3]  # Pass top 3 to Claude
            ])
        else:
            local_events_text = 'No major events found for this date — the city will be relaxed.'

        # Customer name
        if booking.user:
            customer_name = f'{booking.user.first_name} {booking.user.last_name}'
        else:
            customer_name = f'{booking.guest_first_name} {booking.guest_last_name}'

        # Build the user prompt
        prompt = USER_PROMPT_TEMPLATE.format(
            experience_name=experience.name,
            experience_description=experience.description or '',
            duration_hours=experience.duration_hours,
            core_stops=experience.core_stops or 'As described in the experience',
            customer_name=customer_name,
            guest_count=booking.guest_count or 1,
            pickup_city=pickup_city,
            pickup_address=booking.pickup_address or f'{pickup_city}, CA',
            tour_date=tour_date,
            start_time=start_time,
            special_requests=booking.special_requests or 'None',
            local_events_text=local_events_text,
        )

        # 2. Call Claude API
        message = client.messages.create(
            model='claude-sonnet-4-6',
            max_tokens=2048,
            temperature=0.4,
            system=SYSTEM_PROMPT,
            messages=[{'role': 'user', 'content': prompt}],
        )

        raw = message.content[0].text.strip()

        # Strip any accidental markdown fences
        if raw.startswith('```'):
            raw = raw.split('\n', 1)[1].rsplit('```', 1)[0]

        itinerary = json.loads(raw)
        itinerary['generated_at'] = datetime.now(timezone.utc).isoformat()
        itinerary['booking_id']   = booking.booking_id
        itinerary['version']      = 1
        itinerary['local_events_raw'] = local_events  # Store raw event data

        return itinerary

    except Exception as e:
        import logging
        logging.getLogger('itinerary').error(f'Itinerary generation failed for {booking.booking_id}: {e}')
        return _fallback_itinerary(booking)


def _fallback_itinerary(booking) -> dict:
    """
    Minimal fallback used when Claude or event APIs fail.
    Ensures the booking still has a usable itinerary.
    """
    return {
        'itinerary_title': booking.experience.name,
        'greeting': f'We look forward to welcoming you on {booking.timeslot.slot_date}.',
        'pickup': {
            'address': booking.pickup_address or booking.pickup_city,
            'time': booking.timeslot.start_time.strftime('%I:%M %p') if booking.timeslot.start_time else '9:00 AM',
            'note': 'Your guide will be in touch to confirm pickup details.',
        },
        'local_buzz': [],
        'stops': [],
        'return': {
            'estimated_time': 'As per booking duration',
            'drop_off': booking.pickup_address or booking.pickup_city,
        },
        'inclusions': ['Door-to-door pickup', 'Snacks and water on board'],
        'staff_notes': 'Standard pickup — no special requests.',
        'generated_at': datetime.now(timezone.utc).isoformat(),
        'booking_id': booking.booking_id,
        'version': 0,  # 0 = fallback
        'is_fallback': True,
    }

5. Itinerary Generation Triggers
5.1 Trigger 1 — Booking Confirmation
The primary trigger. Immediately after a booking is confirmed (Stripe webhook checkout.session.completed fires), the itinerary generation job is queued asynchronously so it does not block the checkout response.

from app.itinerary.tasks import queue_itinerary_generation

# Inside checkout.session.completed handler, after booking is confirmed:
if booking.booking_status == 'confirmed':
    # ... existing confirmation logic ...

    # Queue itinerary generation (async — does not block webhook response)

# app/itinerary/tasks.py
from app.itinerary.generator import generate_itinerary
from app.itinerary.storage import save_itinerary
from app.models import Booking
import threading

def queue_itinerary_generation(booking_id: str, trigger: str = 'booking_confirmed'):
    """
    Run itinerary generation in a background thread.
    For production, replace threading with Celery or APScheduler task queue.
    """
    def _run():
        from app import create_app
        app = create_app('production')
        with app.app_context():
            booking = Booking.query.get(booking_id)
            if not booking:
                return
            itinerary = generate_itinerary(booking)
            save_itinerary(booking_id, itinerary, trigger=trigger)

    t = threading.Thread(target=_run, daemon=True)
    t.start()
5.2 Trigger 2 — 48-Hour Pre-Tour Refresh
48 hours before the tour date, the system re-fetches local events (some events are only announced days before) and regenerates the itinerary if better local events are found. The new version is stored with version incremented; the old version is archived (is_active=False).
# app/itinerary/refresh.py
from datetime import datetime, timezone, timedelta
from app.models import Booking, Timeslot, BookingItinerary
from app.itinerary.generator import generate_itinerary
from app.itinerary.storage import save_itinerary
from app.extensions import db
from sqlalchemy import cast, Date

def refresh_upcoming_itineraries():
    """
    Called by cron job every 6 hours.
    Finds confirmed bookings with tours 48 hours away and refreshes itineraries.
    """
    now         = datetime.now(timezone.utc)
    refresh_from = now + timedelta(hours=46)  # 46–50 hour window
    refresh_to   = now + timedelta(hours=50)

    upcoming = (
        Booking.query
        .join(Timeslot, Booking.timeslot_id == Timeslot.timeslot_id)
        .filter(
            Booking.booking_status == 'confirmed',
            Timeslot.slot_datetime >= refresh_from,
            Timeslot.slot_datetime <= refresh_to,
        )
        .all()
    )

    refreshed = 0
    for booking in upcoming:
        try:
            itinerary = generate_itinerary(booking)
            save_itinerary(booking.booking_id, itinerary, trigger='48hr_refresh')
            refreshed += 1
        except Exception as e:
            import logging
            logging.getLogger('itinerary').error(
                f'48hr refresh failed for {booking.booking_id}: {e}')

    return refreshed

5.3 Trigger 3 — Staff Briefing Email (Midnight the Night Before)
# app/itinerary/staff.py
from datetime import datetime, timezone, timedelta, date
from app.models import Booking, Timeslot, BookingItinerary
from app.extensions import db, mail
from flask_mail import Message
from flask import url_for
import os

def send_staff_briefings():
    """
    Called by cron at 00:05 UTC each day.
    Sends briefing email to the assigned staff member / provider
    for every confirmed booking with a timeslot scheduled for today.
    """
    today = date.today()

    bookings = (
        Booking.query
        .join(Timeslot, Booking.timeslot_id == Timeslot.timeslot_id)
        .filter(
            Booking.booking_status == 'confirmed',
            Timeslot.slot_date == today,
        )
        .all()
    )

    sent = 0
    for booking in bookings:
        # Determine recipient: assigned staff or provider
        recipient_email = None
        recipient_name  = 'Team'

        if booking.staff_member:
            recipient_email = booking.staff_member.email
            recipient_name  = booking.staff_member.full_name.split()[0]
        elif booking.experience.provider:
            recipient_email = booking.experience.provider.user.email
            recipient_name  = booking.experience.provider.business_name

        if not recipient_email:
            recipient_email = os.environ.get('ADMIN_EMAIL')

        # Build briefing URL (no login required — token secured)
        briefing_url = url_for('itinerary.staff_briefing',
                                booking_id=booking.booking_id,
                                _external=True)

        msg = Message(
            subject=f'[BAE Today] {booking.experience.name} — {booking.pickup_city} pickup',
            recipients=[recipient_email],
        )
        msg.body = (
            f'Good morning {recipient_name},\n\n'
            f'You have a tour today:\n'
            f'  Experience: {booking.experience.name}\n'
            f'  Guest: {booking.guest_first_name} {booking.guest_last_name}'
            f' ({booking.guest_count} guest(s))\n'
            f'  Pickup: {booking.pickup_address or booking.pickup_city}\n'
            f'  Time: {booking.timeslot.start_time.strftime("%I:%M %p")}\n\n'
            f'Full briefing (mobile-optimized):\n{briefing_url}\n\n'
            f'Bay Area Experiences | (408) 831-2101'
        )
        mail.send(msg)

        # Mark briefing as sent on the itinerary record
        itinerary = BookingItinerary.query.filter_by(
            booking_id=booking.booking_id, is_active=True).first()
        if itinerary:
            itinerary.staff_notified_at = datetime.now(timezone.utc)
            db.session.commit()

        sent += 1

    return sent

# Cron (GoDaddy cPanel — runs at 12:05 AM daily):
